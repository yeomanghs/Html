def divide_chunks(l, n):
    # How many elements each
    # list should have
    # looping till length l
    for i in range(0, len(l), n): 
        yield l[i:i + n]

def parse_item(soup):
    itemList = []
    for no, item in enumerate(soup.find_all('table')):
        #get first table
        if no == 0:
            for info in item.find_all('td'):
#                 print(info.get_text().strip())
                allInfo = info.get_text().strip()
                itemList.append(allInfo)
